** Functions for custom bindings
#+begin_src emacs-lisp
  ;; Reload config
  (defun amo/reload-config ()
    "Reload configuration"
    (interactive)
    ;; (dot-env-load)
    (load-file "~/.emacs.d/terminal/init.el"))
#+end_src

** Custom Bindings
#+begin_src emacs-lisp
  :bind (("C-z r" . amo/reload-config)))
#+end_src
